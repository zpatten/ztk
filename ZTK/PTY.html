def spawn(*args, &block)
  begin
    ::PTY.spawn(*args) do |reader, writer, pid|
      begin
        block_given? and yield(reader, writer, pid)
      rescue Errno::EIO
      ensure
        ::Process.wait(pid)
      end
    end
  rescue ::PTY::ChildExited
  end

  true
end
